def average(marks):
    sum=0
    length=0
    for i in marks:
        if (i>0):
            sum=sum+i
            length=length+1
    average=sum/length
    return average
def highest(marks):
    high=0
    for i in marks:
        if (i>high):
            high=i
    return high
def lowest(marks):
    low=100
    for i in marks:
        if (low>i>0):
            low=i
    return low
def absent(marks):
    ab=0
    for i in marks:
        if (i== -1):
            ab=ab+1
    return ab
